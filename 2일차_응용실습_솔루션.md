## 실습 1 — 기울기 조종 메카넘카

### 🎯 1-A. 속도도 기울기로! (전체 코드)

**아이디어** 더 크게 기울인 축의 크기(`max(|x|,|y|)`)를 속도 40~90으로 환산해 `mecanum(code, speed)`에 전달. OLED에 속도(%)를 함께 표시합니다.

```python
# tilt_car_speed.py  ----  도전 1-A
from ai_ponybot import i2c, PonyMotor, PonyOLED
from neopixel import NeoPixel
from machine import Pin
import mpu6050
from time import sleep

imu   = mpu6050.accel(i2c, 0x69)
motor = PonyMotor(i2c)
oled  = PonyOLED(i2c)
np    = NeoPixel(Pin(13, Pin.OUT), 25)

DEAD  = 4000
CODE  = {'fwd': 8, 'back': 2, 'left': 4, 'right': 6, 'stop': 5}
LABEL = {'fwd': 'FWD', 'back': 'BACK', 'left': 'LEFT', 'right': 'RIGHT', 'stop': 'STOP'}
ARROW = {
    'fwd':  [2, 6, 7, 8, 10, 12, 14, 17, 22],
    'back': [2, 7, 10, 12, 14, 16, 17, 18, 22],
    'left': [2, 6, 10, 11, 12, 13, 14, 16, 22],
    'right':[2, 8, 10, 11, 12, 13, 14, 18, 22],
    'stop': [12],
}

def tilt_to_dir(x, y):
    if abs(x) < DEAD and abs(y) < DEAD:
        return 'stop'
    if abs(y) >= abs(x):
        return 'fwd' if y > 0 else 'back'
    return 'right' if x > 0 else 'left'

def tilt_to_speed(x, y):                 # ★ 기울기 크기 -> 속도(40~90)
    mag = max(abs(x), abs(y))
    return min(40 + mag // 400, 90)

def show_arrow(name):
    for i in range(25): np[i] = (0, 0, 0)
    for i in ARROW[name]: np[i] = (0, 10, 0)
    np.write()

print('기울기 속도 조종! (Ctrl+C 정지)')
last_d, last_bucket = None, -1
while True:
    v = imu.get_values()
    d = tilt_to_dir(v['AcX'], v['AcY'])
    if d == 'stop':
        motor.mecanum(5); spd = 0
    else:
        spd = tilt_to_speed(v['AcX'], v['AcY'])   # ★
        motor.mecanum(CODE[d], spd)

    bucket = spd // 10                              # ★ 속도 10단위로 바뀔 때만 갱신(깜빡임 방지)
    if d != last_d or bucket != last_bucket:
        show_arrow(d)
        oled.clear()
        oled.write_line(0, LABEL[d])
        oled.write_line(1, 'SPD: %d%%' % spd)       # ★ 속도 표시
        oled.show()
        last_d, last_bucket = d, bucket
    sleep(0.05)
```

### 🎯 1-B. 대각선 8방향 (전체 코드)

**아이디어** X·Y를 각각 `-1/0/+1`로 분류한 뒤 키패드 코드(7 8 9 / 4 5 6 / 1 2 3)로 매핑. 대각선 LED 화살표(↗↖↘↙)도 추가합니다.

```python
# tilt_car_8way.py  ----  도전 1-B
from ai_ponybot import i2c, PonyMotor, PonyOLED
from neopixel import NeoPixel
from machine import Pin
import mpu6050
from time import sleep

imu   = mpu6050.accel(i2c, 0x69)
motor = PonyMotor(i2c)
oled  = PonyOLED(i2c)
np    = NeoPixel(Pin(13, Pin.OUT), 25)

DEAD = 4000
NAME = {8:'FWD', 2:'BACK', 4:'LEFT', 6:'RIGHT', 9:'F-R', 7:'F-L', 3:'B-R', 1:'B-L', 5:'STOP'}
ARROW8 = {                               # ★ 8방향 + 정지
    8: [2, 6, 7, 8, 10, 12, 14, 17, 22],     # ↑
    2: [2, 7, 10, 12, 14, 16, 17, 18, 22],   # ↓
    4: [2, 6, 10, 11, 12, 13, 14, 16, 22],   # ←
    6: [2, 8, 10, 11, 12, 13, 14, 18, 22],   # →
    9: [4, 8, 12, 16, 20, 3, 9],             # ↗ (앞-우)
    7: [0, 6, 12, 18, 24, 1, 5],             # ↖ (앞-좌)
    3: [0, 6, 12, 18, 24, 19, 23],           # ↘ (뒤-우)
    1: [4, 8, 12, 16, 20, 15, 21],           # ↙ (뒤-좌)
    5: [12],
}

def tilt_to_code(x, y):                  # ★ 8방향 판정
    fx = 1 if x > DEAD else (-1 if x < -DEAD else 0)    # 우(+)/좌(-)
    fy = 1 if y > DEAD else (-1 if y < -DEAD else 0)    # 앞(+)/뒤(-)
    table = {(0, 0): 5,
             (0, 1): 8, (0, -1): 2, (1, 0): 6, (-1, 0): 4,
             (1, 1): 9, (-1, 1): 7, (1, -1): 3, (-1, -1): 1}
    return table[(fx, fy)]

def show_arrow(code):
    for i in range(25): np[i] = (0, 0, 0)
    for i in ARROW8[code]: np[i] = (0, 10, 0)
    np.write()

print('8방향 기울기 조종! (Ctrl+C 정지)')
last = None
while True:
    v = imu.get_values()
    code = tilt_to_code(v['AcX'], v['AcY'])
    motor.mecanum(code, 60)
    if code != last:
        show_arrow(code)
        oled.clear()
        oled.write_line(0, '8-WAY DRIVE')
        oled.write_line(1, NAME[code])
        oled.show()
        last = code
    sleep(0.05)
```

## 실습 2 — 박수 리듬 댄스봇

### 🎯 2-A. 소리 막대 미터 (추가 함수 + 루프 1줄 교체)

**아이디어** 대기·박수 카운트 중에 소리 크기를 `n = level // 80` 개의 LED 막대로 실시간 표시. 단계별 색(초록→노랑→빨강).

```python
# ★ 함수 추가 (실습 2 코드의 함수 정의부에)
def sound_bar(level):
    n = min(level // 80, 25)
    for i in range(25):
        if   i >= n: np[i] = (0, 0, 0)
        elif i < 10: np[i] = (0, 10, 0)     # 낮음 = 초록
        elif i < 18: np[i] = (10, 8, 0)     # 중간 = 노랑
        else:        np[i] = (12, 0, 0)     # 높음 = 빨강
    np.write()

def count_claps_metered(window_ms=2000):    # ★ 막대 표시하며 박수 세기
    count, t0 = 0, ticks_ms()
    while ticks_diff(ticks_ms(), t0) < window_ms:
        lv = sound_level()
        sound_bar(lv)                       # ★ 실시간 막대
        if lv > THRESH:
            count += 1
            sleep(0.3)
        sleep(0.02)
    return count
```

## 실습 3 — 멀티모달 감정 로봇

### 🎯 3-A. 네 번째 감정 추가 — angry (전체 코드)

**아이디어** 버튼 B를 '삐짐' 입력으로 추가. 빨간 표정(눈썹 `0,4` + 찡그린 입) + `Hmph!` + 뒤로 물러나는 반응.

```python
# emotion_bot_angry.py  ----  도전 3-A
from ai_ponybot import i2c, PonyMotor, PonyOLED
from neopixel import NeoPixel
from machine import Pin, ADC
import mpu6050
from time import sleep

imu   = mpu6050.accel(i2c, 0x69)
motor = PonyMotor(i2c)
oled  = PonyOLED(i2c)
np    = NeoPixel(Pin(13, Pin.OUT), 25)
btn_a = Pin(36, Pin.IN)
btn_b = Pin(39, Pin.IN)                 # ★ 삐짐 입력
mic   = ADC(Pin(35)); mic.atten(ADC.ATTN_11DB)

FACE = {
    'happy':    {'eyes': [6, 8],       'mouth': [15, 19, 21, 22, 23], 'c': (0, 12, 0)},
    'surprise': {'eyes': [6, 8],       'mouth': [11, 13, 17, 21, 23], 'c': (12, 8, 0)},
    'dizzy':    {'eyes': [6, 8],       'mouth': [16, 18, 22],          'c': (8, 0, 12)},
    'angry':    {'eyes': [0, 4, 6, 8], 'mouth': [16, 18, 22],          'c': (14, 0, 0)},  # ★ 눈썹+찡그림(빨강)
}
SAY = {'happy': 'I love it!', 'surprise': 'Whoa!', 'dizzy': 'So dizzy...', 'angry': 'Hmph!'}  # ★

def show_face(emo):
    for i in range(25): np[i] = (0, 0, 0)
    f = FACE[emo]
    for i in f['eyes'] + f['mouth']:
        np[i] = f['c']
    np.write()

def read_emotion():
    s = [mic.read() for _ in range(40)]
    if max(s) - min(s) > 1800: return 'surprise'
    if btn_a.value() == 0:     return 'happy'
    if btn_b.value() == 0:     return 'angry'      # ★ 추가
    v = imu.get_values()
    if abs(v['AcX']) > 9000 or abs(v['AcY']) > 9000: return 'dizzy'
    return None

def react(emo):
    if emo == 'happy':
        motor.drive('left', 50); sleep(0.8); motor.drive('stop')
    elif emo == 'surprise':
        motor.mecanum(2, 60); sleep(0.5); motor.mecanum(5)
    elif emo == 'dizzy':
        motor.mecanum(4, 50); sleep(0.3); motor.mecanum(6, 50); sleep(0.3); motor.mecanum(5)
    elif emo == 'angry':                            # ★ 삐져서 뒤로 쿵
        motor.mecanum(2, 70); sleep(0.5); motor.mecanum(5)

# 평상시 얼굴
for i in range(25): np[i] = (0, 0, 0)
np[6] = np[8] = (0, 0, 6); np[16] = np[17] = np[18] = (0, 0, 6); np.write()
oled.clear(); oled.write_line(0, 'EMOTION BOT'); oled.write_line(1, 'A/B/clap/tilt'); oled.show()

while True:
    emo = read_emotion()
    if emo:
        show_face(emo)
        oled.clear(); oled.write_line(0, emo.upper()); oled.write_line(1, SAY[emo]); oled.show()
        react(emo)
        sleep(0.6)
    sleep(0.05)
```

> 참고: `dizzy`와 `angry`는 입 모양이 같지만 **색(보라 vs 빨강)과 눈썹(0,4)** 으로 구분됩니다. 더 확실히 하려면 angry 입을 `[15, 16, 18, 19]` 등으로 바꿔 보세요.

## 실습 5 — 온도 탐사 로버

### 🎯 5-A. 최고온 지점으로 복귀 (전체 코드)

**아이디어** 정거장마다 최고 온도와 그 번호를 기억하고, 종료(B) 시 그 지점까지 **뒤로 이동해 복귀**합니다.

```python
# temp_rover_return.py  ----  도전 5-A
from ai_ponybot import i2c, PonyMotor, PonyOLED
from neopixel import NeoPixel
from machine import Pin
import esp32
from time import sleep

motor = PonyMotor(i2c)
oled  = PonyOLED(i2c)
np    = NeoPixel(Pin(13, Pin.OUT), 25)
btn_a = Pin(36, Pin.IN)
btn_b = Pin(39, Pin.IN)

def read_temp():
    return esp32.mcu_temperature()

def label(diff):
    if   diff <= 1: return 'COOL'
    elif diff <= 3: return 'OK'
    return 'HOT'

def temp_bar(t, base):
    diff = t - base
    level = max(0, min(5, diff))
    col = (0, 0, 12) if diff <= 1 else ((0, 12, 0) if diff <= 3 else (14, 0, 0))
    for i in range(25): np[i] = (0, 0, 0)
    for r in range(level):
        for c in range(5): np[(4 - r) * 5 + c] = col
    np.write()

base = read_temp()
oled.clear(); oled.write_line(0, 'TEMP ROVER'); oled.write_line(1, 'base=%dC A=go' % base); oled.show()
while btn_a.value() == 1:
    sleep(0.05)

station = 0
hot_t, hot_station = -999, 0                 # ★ 최고온 기록
while True:
    station += 1
    motor.mecanum(8, 50); sleep(1.0); motor.mecanum(5); sleep(0.3)

    t = read_temp()
    temp_bar(t, base)
    if t > hot_t:                            # ★ 최고온 갱신
        hot_t, hot_station = t, station
    oled.clear()
    oled.write_line(0, 'Station %d' % station)
    oled.write_line(1, '%dC  %s' % (t, label(t - base)))
    oled.show()
    sleep(1.5)

    if btn_b.value() == 0:                   # 종료 -> 최고온 지점 복귀
        motor.mecanum(5)
        oled.clear(); oled.write_line(0, 'Hottest #%d' % hot_station)
        oled.write_line(1, '%dC back...' % hot_t); oled.show(); sleep(1)
        for _ in range(station - hot_station):   # ★ 뒤로 N칸 이동
            motor.mecanum(2, 50); sleep(1.0); motor.mecanum(5); sleep(0.2)
        oled.clear(); oled.write_line(0, 'Arrived')
        oled.write_line(1, 'Hottest #%d' % hot_station); oled.show()
        break
```

## 정리 — 각 도전과제가 가르치는 것

| 도전 | 핵심 개념 |
|------|-----------|
| 1-A | 센서값 → 연속 제어량 매핑(속도), OLED 갱신 throttle |
| 1-B | 2축 입력 → 다중 상태 분기(키패드 코드), 좌표 LED 패턴 설계 |
| 2-A | 실시간 시각화(막대 미터), 단계별 색 |
| 2-B | 임계 이벤트 처리·래치(latch)·재개 조건 |
